  push:
    branches: [ main ]
  pull_request:
    branches: [ main ]

jobs:
  build-and-deploy:
    runs-on: ubuntu-latest   # The virtual machine environment

    steps:
      # ğŸ§± Step 1: Checkout repository code
      - name: Checkout code
        uses: actions/checkout@v4

      # âš™ï¸ Step 2: Set up Node.js
      - name: Set up Node.js
        uses: actions/setup-node@v4
        with:
          node-version: '20'     # Node.js version
          cache: 'npm'

      # ğŸ“¦ Step 3: Install project dependencies
      - name: Install dependencies
        run: npm ci

      # ğŸ§ª Step 4: Run automated tests
      - name: Run tests
        run: npm test -- --watch=false

      # ğŸ—ï¸ Step 5: Build the project (for example React build)
        # For React: npm run build
        # For Node API: npm run build (optional)
      - name: Build project
        run: npm run build

      # ğŸ“¤ Step 6: Upload build artifacts (for verification)
      - name: Upload build artifacts
        uses: actions/upload-artifact@v4
        with:
          name: build-files
          path: build/

      # ğŸŒ Step 7 (Optional): Deploy to GitHub Pages
      - name: Deploy to GitHub Pages

          github_token: ${{ secrets.GITHUB_TOKEN }}
          publish_dir: ./build
